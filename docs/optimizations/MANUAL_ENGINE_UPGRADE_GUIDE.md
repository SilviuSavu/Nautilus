        if [[ "$m4_max" == "true" ]]; then
            echo "   üöÄ M4 Max acceleration: ACTIVE"
        else
            echo "   ‚ö†Ô∏è M4 Max acceleration: Not detected"
        fi
    else
        echo "‚ùå $engine_name: FAILED or NOT RESPONDING"
    fi
    
    echo ""
done

echo "üìà UPGRADE RESULTS:"
echo "   Healthy engines: $healthy_count/$total_engines"
echo "   Success rate: $(( healthy_count * 100 / total_engines ))%"

    echo "üéâ MANUAL UPGRADE COMPLETE! All engines operational."
